    //COMANDO PARA CREAR UNA RAMA  (Chicos pongan su nombre)
    git checkout -b nombre-de-la-rama

    //COMANDO PARA AGREGAR TODOS LOS CAMBIOS AL STAGE
    git add . 

    //COMANDO PARA AGREGAR UN SOLO ARCHIVO AL STAGE
    git add nombre-archivo.extencion

    //COMANDO PARA HACER UN COMMIT
    git commit -m "INGRESA EL TEXTO DE TU COMIT"

    //COMANDO PARA HACER PUSH A SU RAMA
    git push origin nombre-de-la-rama

    //COMANDO PARA BAJARSE LOS CAMBIOS, LOS CAMBIOS SI SE BAJAN DE LA RAMA MAIN/MASTER PORQUE AQUI SE INTEGRAN
    git pull origin

    //COMANDO PARA REGRESAR A LA RAMA MASTER
    git checkout main

    //COMANDO PARA CLONAR UN REPOSITORIO REMOTO: Este comando se usa para crear una copia local de un repositorio remoto en tu máquina.
    git clone <URL_del_repositorio>


    //COMANDO PARA CAMBIAR A UNA RAMA EXISTENTE:Cambia a una rama existente.
    git checkout nombre-de-la-rama

    //COMANDO PARA VER EL ESTADO DE LOS ARCHIVOS:Muestra el estado de los archivos en tu directorio de trabajo.
    git status

    //COMANDO PARA AGREGAR CAMBIOS AL ÁREA DE STAGING:Agrega todos los cambios en el directorio de trabajo al área de staging.
    git add .

    //COMANDO PARA AGREGAR UN SOLO ARCHIVO AL ÁREA DE STAGING:Agrega un archivo específico al área de staging.
    git add nombre-archivo.extension

    //COMANDO PARA HACER UN COMMIT: Crea un commit con los cambios en el área de staging y agrega un mensaje descriptivo.
    git commit -m "INGRESA EL TEXTO DE TU COMMIT"

    //COMANDO PARA HACER PUSH A UNA RAMA REMOTA:Sube tus cambios a la rama remota especificada.
    git push origin nombre-de-la-rama

    //COMANDO PARA DESCARGAR CAMBIOS DE UNA RAMA REMOTA:Actualiza tu rama local con los cambios de la rama remota.
    git pull origin nombre-de-la-rama

    //COMANDO PARA REGRESAR A LA RAMA PRINCIPAL:Cambia a la rama principal (generalmente 'main' o 'master').
    git checkout main

    //COMANDO PARA VER EL HISTORIAL DE COMMITS:Muestra el historial de commits en la rama actual.
    git log

    //COMANDO PARA VER LAS RAMAS LOCALES Y REMOTAS:Muestra una lista de todas las ramas locales y remotas.
    git branch -a

    //COMANDO PARA FUSIONAR CAMBIOS DE OTRA RAMA EN LA RAMA ACTUAL:Combina los cambios de otra rama en la rama actual.
    git merge nombre-de-la-otra-rama

    //COMANDO PARA ELIMINAR UNA RAMA:Elimina una rama local después de que se hayan fusionado los cambios.
    git branch -d nombre-de-la-rama

    //COMANDO PARA VER DIFERENCIAS ENTRE ARCHIVOS:Muestra las diferencias entre dos commits, ramas o archivos.
    git diff origen..destino

    //COMANDO PARA VER CAMBIOS ANTES DE HACER UN COMMIT:Muestra las diferencias entre el directorio de trabajo y el área de staging.
    git diff

    //Eliminar un archivo del área de staging:Si deseas quitar un archivo del área de staging (sin eliminarlo), puedes utilizar:

    git reset nombre-archivo.extension

    //Descartar cambios en el directorio de trabajo:Si deseas deshacer los cambios realizados en un archivo (antes de hacer commit):
    git checkout nombre-archivo.extension

    //Eliminar archivos del repositorio y del sistema de archivos:Para eliminar archivos del repositorio y del sistema de archivos:
    git rm nombre-archivo.extension
